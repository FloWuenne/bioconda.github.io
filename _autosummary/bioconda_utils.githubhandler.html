    strict: False        # require PR branch to be up to date with base

    dismissal_restrictions:  # specify who may dismiss reviews

         - login: bla
       teams:
         - id: 1
         - name: Bl Ub
         - slug: bl-ub
restrictions:             # specify who may push
  users:
    - login: bla
  teams:
    - id: 1
